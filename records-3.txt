interface Printable {
    void printDetails();
}

public record User(String username, String email) implements Printable {
    @Override
    public void printDetails() {
        System.out.println("Username: " + username + ", Email: " + email);
    }
}

public class MainWithInterface {
    public static void main(String[] args) {
        User user1 = new User("johndoe", "john.doe@example.com");
        user1.printDetails();
    }
}
